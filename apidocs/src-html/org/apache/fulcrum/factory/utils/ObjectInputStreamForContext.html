022import java.io.InputStream;
023import java.io.IOException;
024import java.io.ObjectInputStream;
025import java.io.ObjectStreamClass;
026
027/**
028 * A deserialization stream for a specific class loader context.
029 *
030 * @author <a href="mailto:ilkka.priha@simsoft.fi">Ilkka Priha</a>
031 * @version $Id$
032 */
033public class ObjectInputStreamForContext extends ObjectInputStream
034{
035    /**
036     * The class loader of the context.
037     */
038    private ClassLoader classLoader;
039
040    /**
041     * Required to make satisfy the proxy methods
042     * 
043     * @throws IOException Generic exception
044     */
045    public ObjectInputStreamForContext()
046        throws IOException
047    {
048    }
049
050    /**
051     * This method will construct a new object stream for a context.
052     *
053     * @param in the serialized input stream.
054     * @param loader the class loader of the context.
055     * @throws IOException on errors.
056     */
057    public  ObjectInputStreamForContext(InputStream in,
058                                        ClassLoader loader)
059                                        throws IOException
060    {
061        super(in);
062        classLoader = loader;
063    }
064
065    /**
066     * {@link java.io.ObjectInputStream#resolveClass(ObjectStreamClass)}
067     * 
068     * @param v ObjectStreamClass to resolve
069     * @return {@inheritDoc} class to resolve
070     * @throws IOException if object stream not found
071     * @throws ClassNotFoundException if class not found
072     */
073    protected Class<?> resolveClass(ObjectStreamClass v)
074                                 throws IOException,
075                                 ClassNotFoundException
076    {
077        return classLoader == null ?
078            super.resolveClass(v) : classLoader.loadClass(v.getName());
079    }
080}
